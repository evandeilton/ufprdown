\else
\newenvironment{CSLReferences}[2] % #1 hanging-indent, #2 entry-spacing

  \ifodd #1
    \setlength{\leftskip}{\cslhangindent}%
    \setlength{\parindent}{-\cslhangindent}%
  \fi}
 {\par}
\fi
